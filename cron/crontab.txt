# Situation Room — production cron jobs
# Install with: crontab cron/crontab.txt
# Or append to existing: crontab -l | cat - cron/crontab.txt | crontab -
#
# All routes protected by CRON_SECRET (Authorization: Bearer $CRON_SECRET)
# Replace https://situationroom.space with your domain if different.

CRON_SECRET=your-cron-secret-here
SITE=https://situationroom.space

# Daily data refresh — UTXO age at 06:00 UTC
0 6 * * * curl -s -H "Authorization: Bearer $CRON_SECRET" $SITE/api/data/utxo-age >> /opt/situationroom-v3/logs/cron.log 2>&1

# Weekly data — CB assets + rates every Monday 06:05 UTC
5 6 * * 1 curl -s -H "Authorization: Bearer $CRON_SECRET" $SITE/api/data/cbassets >> /opt/situationroom-v3/logs/cron.log 2>&1
5 6 * * 1 curl -s -H "Authorization: Bearer $CRON_SECRET" $SITE/api/data/cbrates >> /opt/situationroom-v3/logs/cron.log 2>&1

# Daily briefing — 06:00 UTC
0 6 * * * curl -s -H "Authorization: Bearer $CRON_SECRET" $SITE/api/cron/daily-briefing >> /opt/situationroom-v3/logs/cron.log 2>&1

# Daily snapshot — 06:05 UTC (after briefing)
5 6 * * * curl -s -H "Authorization: Bearer $CRON_SECRET" $SITE/api/cron/daily-snapshot >> /opt/situationroom-v3/logs/cron.log 2>&1

# Payment confirmation polling — every 5 minutes
*/5 * * * * curl -s -H "x-cron-secret: $CRON_SECRET" $SITE/api/payments/confirm >> /opt/situationroom-v3/logs/cron.log 2>&1

# SitRoom AI chat alerts — every 5 minutes
*/5 * * * * curl -s -H "Authorization: Bearer $CRON_SECRET" $SITE/api/cron/chat-alerts >> /opt/situationroom-v3/logs/cron.log 2>&1

# Subscription expiry check — daily 07:00 UTC
0 7 * * * curl -s -H "Authorization: Bearer $CRON_SECRET" $SITE/api/cron/subscription-expiry >> /opt/situationroom-v3/logs/cron.log 2>&1

# Free newsletter digest — Sunday 18:00 UTC
0 18 * * 0 curl -s -H "Authorization: Bearer $CRON_SECRET" $SITE/api/cron/newsletter-digest >> /opt/situationroom-v3/logs/cron.log 2>&1

# General/Members daily briefing email — 06:15 UTC (15 min after briefing generation)
15 6 * * * curl -s -H "Authorization: Bearer $CRON_SECRET" $SITE/api/cron/newsletter-daily >> /opt/situationroom-v3/logs/cron.log 2>&1

# M2 money supply data refresh — weekly Monday 06:10 UTC
10 6 * * 1 curl -s $SITE/api/data/m2 >> /opt/situationroom-v3/logs/cron.log 2>&1

# VIP personalised briefings — 06:10 UTC (after base briefing confirmed)
10 6 * * * curl -s -H "Authorization: Bearer $CRON_SECRET" $SITE/api/cron/vip-briefings >> /opt/situationroom-v3/logs/cron.log 2>&1

# User alert checks — every 5 minutes
*/5 * * * * curl -s -H "Authorization: Bearer $CRON_SECRET" $SITE/api/cron/check-alerts >> /opt/situationroom-v3/logs/cron.log 2>&1

# AI Trading — 4h cycle at 00:05, 04:05, 08:05, 12:05, 16:05, 20:05 UTC
5 0,4,8,12,16,20 * * * curl -s -H "Authorization: Bearer $CRON_SECRET" $SITE/api/cron/trading-cycle >> /opt/situationroom-v3/logs/cron.log 2>&1

# Threat score monitor — every 60s, triggers AI analysis on state changes
* * * * * curl -s $SITE/api/data/threat-score >> /opt/situationroom-v3/logs/cron.log 2>&1

# Position sync — every 60s, detects TP/SL hits and liquidations
* * * * * curl -s -H "Authorization: Bearer $CRON_SECRET" $SITE/api/cron/position-sync >> /opt/situationroom-v3/logs/cron.log 2>&1

# Country data refresh — 1st of every month at 03:00 UTC
0 3 1 * * curl -s -H "Authorization: Bearer $CRON_SECRET" $SITE/api/cron/refresh-country-data >> /opt/situationroom-v3/logs/cron.log 2>&1
